function createWindow() {
  const win = new BrowserWindow({
    width: 1380,

ipcMain.handle('scan:url', async (_event, payload) => {
  try {
    return await runUrlScan(payload?.url || '', { auth: payload?.auth || {} });
  } catch (error) {
    return { ok: false, error: error?.message || 'URL scan failed', findings: [], metadata: {} };
  }
});

ipcMain.handle('scan:project', async (_event, payload) => {
  try {
    return await runProjectScan(payload?.folderPath || '');
  } catch (error) {
    return { ok: false, error: error?.message || 'Project scan failed', findings: [], metadata: {} };

    return { ok: true, data: getChecklist() };
  } catch (error) {

  }
});

ipcMain.handle('dialog:pickFolder', async () => {
  const result = await dialog.showOpenDialog({ properties: ['openDirectory'] });
  if (result.canceled || !result.filePaths?.length) return { ok: false };
  return { ok: true, folderPath: result.filePaths[0] };
});

ipcMain.handle('docs:open', async (_event, url) => {
  try {
    await shell.openExternal(url);
    return { ok: true };
  } catch (error) {
    return { ok: false, error: error?.message };
  }
});

ipcMain.handle('report:export', async (_event, payload) => {
  try {
    const format = payload?.format === 'json' ? 'json' : 'html';
    const scanResult = payload?.scanResult;
    if (!scanResult) throw new Error('KhÃ´ng cÃ³ dá»¯ liá»‡u scan Ä‘á»ƒ export.');

    const content = format === 'json' ? buildJsonReport(scanResult) : buildHtmlReport(scanResult);
    const saveResult = await dialog.showSaveDialog({
      title: 'Export report',
      defaultPath: getSuggestedFilename(scanResult, format),
      filters: format === 'json'
        ? [{ name: 'JSON', extensions: ['json'] }]
        : [{ name: 'HTML', extensions: ['html'] }]
    });

    if (saveResult.canceled || !saveResult.filePath) return { ok: false, canceled: true };
    await fs.writeFile(saveResult.filePath, content, 'utf8');
    return { ok: true, filePath: saveResult.filePath };
  } catch (error) {
    return { ok: false, error: error?.message || 'Export failed' };
  }
});

function extractForms(html) {
  const forms = [...(html || '').matchAll(/<form\b[^>]*>([\s\S]*?)<\/form>/gi)];
  return forms.map((m) => m[0]);
}

function extractLinks(html, baseOrigin) {
  const matches = [...(html || '').matchAll(/<a\b[^>]*href=["']([^"'#]+)["']/gi)];
  const links = new Set();
  for (const m of matches) {
    const href = m[1];
    if (/^https?:\/\//i.test(href) || href.startsWith('/')) {
      if (href.startsWith('/')) links.add(`${baseOrigin}${href}`);
      else links.add(href);
    }
  }
  return [...links].slice(0, 30);
}

module.exports = { extractForms, extractLinks };

const { fetchWithMeta, buildRequestHeaders } = require('../../utils/http');

async function probeOptions(url, auth = {}) {
  try {
    const response = await fetch(url, { method: 'OPTIONS', headers: buildRequestHeaders(auth) });
    return { allow: response.headers.get('allow') || '', status: response.status };
  } catch {
    return { allow: '', status: 0 };
  }
}

async function probeMissingPath(origin, auth = {}) {
  try {
    const badUrl = `${origin.replace(/\/$/, '')}/this-path-should-not-exist-security-scan`;
    const result = await fetchWithMeta(badUrl, { headers: buildRequestHeaders(auth) });
    return { url: badUrl, ...result };
  } catch {
    return null;
  }
}

async function probeVariant(url, variant, auth = {}) {
  return fetchWithMeta(url, {
    headers: { ...buildRequestHeaders(auth), ...(variant.headers || {}) },
    method: variant.method || 'GET'
  });
}

module.exports = { probeOptions, probeMissingPath, probeVariant };

===== FILE: engine\collectors\source\project-loader.js =====
const fs = require('fs');
const path = require('path');

const SKIP_DIRS = new Set(['node_modules', '.git', 'dist', 'build', 'bin', 'obj', '.vs']);

function walkFiles(root, maxFiles = 500) {
  const out = [];
  function walk(dir) {
    if (out.length >= maxFiles) return;
    for (const entry of fs.readdirSync(dir, { withFileTypes: true })) {
      if (entry.isDirectory()) {
        if (SKIP_DIRS.has(entry.name)) continue;
        walk(path.join(dir, entry.name));
      } else {
        out.push(path.join(dir, entry.name));
        if (out.length >= maxFiles) return;
      }
    }
  }
  walk(root);
  return out;
}

function readTextSafe(filePath, maxBytes = 250000) {
  try {
    const stat = fs.statSync(filePath);
    if (stat.size > maxBytes) return '';
    return fs.readFileSync(filePath, 'utf8');
  } catch {
    return '';
  }
}

module.exports = { walkFiles, readTextSafe };

  const summary = { total: findings.length, byCategory: {}, bySeverity: {} };

}

async function probeRoutesEnhanced(origin, requestHeaders) {
  const surfaceStatus = {};
  const probePromises = PROBE_ROUTES.map(async route => {
    try {
      const controller = new AbortController();
      const timeout = setTimeout(() => controller.abort(), 3000);
      const probe = await fetch(`${origin}${route}`, {
        redirect: 'manual',
        headers: requestHeaders,
        signal: controller.signal
      });
      clearTimeout(timeout);
      const location = probe.headers.get('location') || '';
      const redirectedToLogin = probe.status >= 300 && probe.status < 400 && /login|signin|account|auth/i.test(location);
      surfaceStatus[route] = {
        status: probe.status,
        redirectedToLogin,
        location,
        server: probe.headers.get('server') || '',
        contentType: probe.headers.get('content-type') || ''
      };
    } catch {
      surfaceStatus[route] = { status: 0, redirectedToLogin: false, location: '' };
    }
  });
  for (let i = 0; i < probePromises.length; i += 5) {
    await Promise.all(probePromises.slice(i, i + 5));
  }
  return surfaceStatus;
}

  const headers = response.headers;

  const setCookies = headers.getSetCookie
    ? headers.getSetCookie()

  const forms = extractForms(text);
  const links = extractLinks(text, parsed.origin);
  const authHints = detectAuthHints(text, toHeaderObject(headers));
  const optionsProbe = await probeOptions(parsed.toString(), auth);

  } : null;

  const contentType = headers.get('content-type') || '';
  const hostname = parsed.hostname.toLowerCase();
  const isLocalhost = hostname === 'localhost' || hostname === '127.0.0.1' || hostname === '::1' || hostname.endsWith('.local');

  const cookieFlags = setCookies.map(raw => {
    const lower = raw.toLowerCase();
    return {
      raw,
      httpOnly: lower.includes('httponly'),
      secure: lower.includes('secure'),
      sameSite: lower.includes('samesite=strict') ? 'Strict' : lower.includes('samesite=lax') ? 'Lax' : lower.includes('samesite=none') ? 'None' : ''
    };
  });

  const surfaceStatus = await probeRoutesEnhanced(parsed.origin, requestHeaders);
  const fingerprint = extractServerFingerprint(headers);

  const context = {
    scannedUrl: parsed.toString(),
    finalUrl,

    status: response.status,

const DEFAULT_HEADERS = {
  'User-Agent': 'OWASP2025Workbench/0.3',
  'Accept': 'text/html,application/xhtml+xml,application/json;q=0.9,*/*;q=0.8'
};

function parseCustomHeaders(customHeaders) {
  if (!customHeaders) return {};
  if (typeof customHeaders === 'object' && !Array.isArray(customHeaders)) return customHeaders;
  if (typeof customHeaders !== 'string') throw new Error('Custom headers pháº£i lÃ  chuá»—i JSON hoáº·c object.');
  const trimmed = customHeaders.trim();
  if (!trimmed) return {};
  try {
    const parsed = JSON.parse(trimmed);
    if (!parsed || Array.isArray(parsed) || typeof parsed !== 'object') throw new Error('Custom headers pháº£i lÃ  JSON object.');
    return parsed;
  } catch (error) {
    throw new Error(`Custom headers JSON khÃ´ng há»£p lá»‡: ${error.message}`);
  }
}

function buildRequestHeaders(auth = {}) {
  const headers = { ...DEFAULT_HEADERS };
  const custom = parseCustomHeaders(auth.customHeaders);
  Object.assign(headers, custom);
  if (auth.cookie?.trim()) headers.Cookie = auth.cookie.trim();
  if (auth.bearerToken?.trim()) headers.Authorization = `Bearer ${auth.bearerToken.trim()}`;
  else if (auth.authorization?.trim()) headers.Authorization = auth.authorization.trim();
  return headers;
}

function maskValue(value = '') {
  if (!value) return '';
  if (value.length <= 8) return '********';
  return `${value.slice(0, 4)}â€¦${value.slice(-3)}`;
}

function summarizeAuth(auth = {}) {
  return {
    hasCookie: Boolean(auth.cookie?.trim()),
    hasBearerToken: Boolean(auth.bearerToken?.trim()),
    hasAuthorization: Boolean(auth.authorization?.trim()),
    hasCustomHeaders: Boolean(auth.customHeaders && String(auth.customHeaders).trim()),
    customHeaderKeys: Object.keys(parseCustomHeaders(auth.customHeaders || {})),
    maskedAuthorization: auth.authorization?.trim() ? maskValue(auth.authorization.trim()) : '',
    maskedBearerToken: auth.bearerToken?.trim() ? maskValue(auth.bearerToken.trim()) : ''
  };
}

async function fetchWithMeta(url, init = {}) {
  try {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 15000);
    try {
      const response = await fetch(url, {
        redirect: 'follow',
        headers: { ...DEFAULT_HEADERS, ...(init.headers || {}) },
        signal: controller.signal,
        ...init
      });
      clearTimeout(timeout);
      const text = await response.text();
      return { response, text, finalUrl: response.url };
    } finally {
      clearTimeout(timeout);
    }
  } catch (error) {
    if (error.name === 'AbortError') throw new Error('QuÃ©t URL timeout (15s)');
    const msg = error.message || '';
    if (msg.includes('ENOTFOUND') || msg.includes('fetch failed')) throw new Error('KhÃ´ng thá»ƒ resolve domain');
    if (msg.includes('ECONNREFUSED')) throw new Error('Káº¿t ná»‘i bá»‹ tá»« chá»‘i');
    if (msg.includes('ETIMEDOUT')) throw new Error('Káº¿t ná»‘i timeout');
    throw new Error(`Lá»—i quÃ©t URL: ${msg}`);
  }
}

function toHeaderObject(headers) {
  return Object.fromEntries(headers.entries());
}
